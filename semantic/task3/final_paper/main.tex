\begin{algorithm}[t]
  \caption{match \textsc{shipInfo} and \textsc{healthInfo} V1 \label{alg:sketch}}
  \begin{algorithmic}[0]
  \small
  \Require $e$: ele shipInfo (name, address)
  \FUNCDESCB $h$: healthInfo (name, address)
  \FUNCDESCB $e.name$: name $\Rightarrow$ (tradeMark,subbranch)
  \Ensure $r$: Match result

  \Function{matchEleV1}{$e, h$}
  \If{$e.name$ match $h.name$}\hfill{\AlgComment{match $\neq$ equal}}
  \State{\Return{$\argmax$(LCS($e.address$, $h.address$))}}
  \EndIf
  \If{$e.address$ match $h.address$}\hfill
  \State{\Return{$\argmax$(LCS($e.name$, $h.name$))}}
  \EndIf 

  \State{// if not match, \bf{do word segment,}}
  \State{$e.nameSeg=seg(e.name)$,}
  \State{$h.nameSeg=seg(h.name)$, }
  \State{$e.addrSeg=seg(e.address)$, }
  \State{$h.addrSeg=seg(h.address)$ }
  \If{match($e.nameSeg$, $h.nameSeg$) \bf{and} \\ match($e.addrSeg$, $h.addrSeg$)}
  \State{\Return{$\argmax$(LCS($e.address$, $h.address$))}}
  \EndIf
  \State{\Return{$r$}}
  \EndFunction
  \normalsize
  \end{algorithmic}
  \end{algorithm}
